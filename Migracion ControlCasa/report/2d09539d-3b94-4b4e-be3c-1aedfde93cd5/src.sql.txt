CREATE TABLE `casa` (
  `idCasa` int(11) NOT NULL AUTO_INCREMENT,
  `calleCasa` varchar(200) CHARACTER SET utf8 NOT NULL,
  `numeroCasa` varchar(30) CHARACTER SET utf8 NOT NULL,
  `precioCasa` float NOT NULL,
  `coloniaCasa` int(11) NOT NULL,
  `ciudadCasa` int(11) NOT NULL,
  `tipoInmuebleCasa` int(11) NOT NULL,
  `tipoTransaccionCasa` int(11) NOT NULL,
  `formaCasa` int(11) NOT NULL,
  `codigoPostalCasa` varchar(10) CHARACTER SET utf8 DEFAULT NULL,
  `propietarioCasa` varchar(30) CHARACTER SET utf8 DEFAULT NULL,
  `nivelesCasa` int(11) DEFAULT NULL,
  `recamaraCasa` int(11) DEFAULT NULL,
  `cocheraCasa` int(11) DEFAULT NULL,
  `banoCasa` float DEFAULT NULL,
  `terrenoCasa` float DEFAULT NULL,
  `construccionCasa` float DEFAULT NULL,
  `fotografiaCasa` mediumblob,
  `descripcionCasa` varchar(400) CHARACTER SET utf8 DEFAULT NULL,
  `frenteCasa` float DEFAULT NULL,
  `fondoCasa` float DEFAULT NULL,
  `esquinaCasa` tinyint(1) DEFAULT NULL,
  `jardinCasa` tinyint(1) DEFAULT NULL,
  `estudioCasa` tinyint(1) DEFAULT NULL,
  `albercaCasa` tinyint(1) DEFAULT NULL,
  `edadCasa` int(11) DEFAULT NULL,
  `telefonoCasa` varchar(15) CHARACTER SET utf8 DEFAULT NULL,
  `enVentaCasa` tinyint(1) DEFAULT NULL,
  PRIMARY KEY (`idCasa`),
  KEY `coloniaCas` (`coloniaCasa`),
  KEY `ciudadCas` (`ciudadCasa`),
  KEY `tipoTras` (`tipoTransaccionCasa`),
  KEY `tipoIn` (`tipoInmuebleCasa`),
  KEY `formaCa` (`formaCasa`),
  CONSTRAINT `ciudadCas` FOREIGN KEY (`ciudadCasa`) REFERENCES `ciudad` (`idCiudad`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `coloniaCas` FOREIGN KEY (`coloniaCasa`) REFERENCES `colonia` (`idColonia`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `formaCa` FOREIGN KEY (`formaCasa`) REFERENCES `forma` (`idForma`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `tipoIn` FOREIGN KEY (`tipoInmuebleCasa`) REFERENCES `tipoinmueble` (`idTipo`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `tipoTras` FOREIGN KEY (`tipoTransaccionCasa`) REFERENCES `tipotransaccion` (`idTipoTransaccion`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=285 DEFAULT CHARSET=utf8 COLLATE=utf8_spanish2_ci
